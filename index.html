⚡ Einmalige Einrichtung — Supabase Tabelle erstellen

Damit Daten zwischen Laptop und Handy synchronisiert werden, führe diesen SQL-Befehl einmal in deinem Supabase Dashboard aus:

  1. Gehe zu supabase.com → dein Projekt
  2. Klicke auf SQL Editor (linke Seitenleiste)
  3. Kopiere den Code unten und klicke Run
-- Create candidates table
CREATE TABLE IF NOT EXISTS candidates (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  phone TEXT NOT NULL,
  language TEXT NOT NULL,
  email TEXT,
  sector TEXT,
  who_type TEXT,
  goal_type TEXT,
  description TEXT,
  status TEXT DEFAULT 'looking',
  cv_url TEXT,
  cv_name TEXT,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Enable Row Level Security
ALTER TABLE candidates ENABLE ROW LEVEL SECURITY;

-- Allow all operations with anon key (admin controls access via password)
CREATE POLICY "Allow all for anon" ON candidates
  FOR ALL USING (true) WITH CHECK (true);

-- Create storage bucket for CVs
INSERT INTO storage.buckets (id, name, public)
VALUES ('cvs', 'cvs', true)
ON CONFLICT DO NOTHING;

-- Allow CV uploads
CREATE POLICY "Allow CV uploads" ON storage.objects
  FOR ALL USING (bucket_id = 'cvs') WITH CHECK (bucket_id = 'cvs');

Nach dem Ausführen: Seite neu laden — das Dashboard wird automatisch funktionieren.
